```java
class Demo
{
	public static void main(String[] args)
	{
		try
		{
			showExce();
			System.out.println("A");
		}
		catch (Exception e)
		{
			System.out.println("B");
		}
		finally
		{
			System.out.println("C");
		}
	}

	public static void showExce() extends Exception
	{
		throw new Exception();
	}
}
```
